\end{DoxyItemize}\hypertarget{md__r_e_a_d_m_e_autotoc_md2}{}\doxysubsection{Usage}\label{md__r_e_a_d_m_e_autotoc_md2}
LP2R \mbox{[}-\/L\mbox{]} \mbox{[}-\/d\mbox{]} \mbox{[}-\/i inputfile\mbox{]} \mbox{[}-\/r resourcefile\mbox{]} \mbox{[}-\/h\mbox{]} \mbox{[}--version\mbox{]} \begin{DoxyVerb}*-L  : Output debugging and extra information in a file named __LP2Rlog.txt__ *

*-d  : Calculate dielectric response assuming type-A dipoles *

*-i inputfile : material and output frequency information supplied via a named file. Default inputfile name is __inp.dat__ * 

*-r resourcefile : Presumably chemistry independent parameters, and output options can be set via a resource file. Default resourcefile name is __LP2R.rc__

*-h : help

*--version : version information
\end{DoxyVerb}
